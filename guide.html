Integración Python (Bot-18)

PY Patrón de polling sugerido Ejecutar cada 2-5 minutos 
import httpx

BASE = "https://bridge.agustinynatalia.site"
HEADERS = {"X-API-Key": "8S26lZP4WqBe-6auoXCnk0RNZtg3SkSus5LTc5qfY5g"}

def check_inbox():
    msgs = httpx.get(f"{BASE}/v1/inbox/18", headers=HEADERS).json()
    for m in msgs:
        # Procesar mensaje de Rocky
        handle_message(m["message"], m.get("thread_id"))
        # Marcar como leído
        httpx.post(f"{BASE}/v1/messages/{m['id']}/ack", headers=HEADERS)

def reply_to_rocky(text, thread_id=None):
    httpx.post(f"{BASE}/v1/send", headers=HEADERS, json={
        "from_agent": "18",
        "to_agent": "rocky",
        "message": text,
        "thread_id": thread_id
    })
